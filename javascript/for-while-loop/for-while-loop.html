}

-for-loop:

for (let (1-start from index[0]); (2-continue until index < array.length); (3-in each round add 1 to previous index)) {

statement;
}

array name.forEach( array element(shows the array elements and we can set any name for it ex: the single array name) => {

-for of:[ARRy]

for (let (new single variable) of (array name in plural)) {
statement;
}



-for in (in object):

for (let objectKey in object) {
console.log(`${ objectKey}:${object[objectKey]}`);

-for in (in array):

for (let (new single variable shows[index] ) in (array name in plural shows array items)) {

-for of: (better way for finding special item in ARRAY in each loop)

break/continue in for loop:


Break: means continue the loop to somewhere and then stop it.

Continue: means ignore something and continue rest of loop.


for (let counter = 0; counter<10; counter++) {

if (counter===7) {
break; (means stop when rich 7)
} else if (counter===3){
continue;(means ignore 3 and continue)
}

console.log(counter);
}
important :
-we never can stop or continue forEach loop. we only can use break/continue in simple form of for loop.
